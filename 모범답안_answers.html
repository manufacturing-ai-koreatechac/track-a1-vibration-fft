import numpy as np
import matplotlib
matplotlib.rcParams['font.family'] = 'AppleGothic'
matplotlib.rcParams['axes.unicode_minus'] = False
import matplotlib.pyplot as plt
from scipy.fft import fft, fftfreq
from sklearn.ensemble import IsolationForest

# 합성 진동 신호 생성 (정상 + 베어링 결함 성분)
np.random.seed(42)
sr = 1000
n = 10000
t = np.linspace(0, n/sr, n)
signal = (0.5 * np.sin(2*np.pi*30*t) +
          0.3 * np.sin(2*np.pi*120*t) +
          0.1 * np.random.randn(n))
# 후반부 고장 성분 강화
fault = n * 3 // 4
signal[fault:] += 0.8 * np.sin(2*np.pi*120*t[fault:]) + 0.3*np.random.randn(n-fault)

print(f"신호 샘플 수: {n}, 샘플링 레이트: {sr} Hz")
print(f"신호 통계 — 평균: {signal.mean():.4f}, 표준편차: {signal.std():.4f}")

신호 샘플 수: 10000, 샘플링 레이트: 1000 Hz
신호 통계 — 평균: -0.0012, 표준편차: 0.5840

과제 A (필수): threshold=2 vs threshold=3 비교

In [2]:
def detect_anomalies(sig, threshold=3):
    mu, sigma = sig.mean(), sig.std()
    return np.abs(sig - mu) > threshold * sigma

anom2 = detect_anomalies(signal, threshold=2)
anom3 = detect_anomalies(signal, threshold=3)

print(f"threshold=2: 이상 포인트 수 = {anom2.sum()}")
print(f"threshold=3: 이상 포인트 수 = {anom3.sum()}")
print(f"→ threshold를 낮추면 이상 포인트가 {anom2.sum()-anom3.sum()}개 더 감지됨")

fig, axes = plt.subplots(2, 1, figsize=(12, 6), sharex=True)
for ax, anom, thr, c in zip(axes, [anom2, anom3], [2, 3], ['crimson','steelblue']):
    ax.plot(t[:2000], signal[:2000], 'gray', lw=0.5, alpha=0.6, label='신호')
    ax.scatter(t[:2000][anom[:2000]], signal[:2000][anom[:2000]],
               color=c, s=20, zorder=5, label=f'이상 ({anom.sum()}개 전체)')
    mu, sigma = signal.mean(), signal.std()
    ax.axhline(mu + thr*sigma, color=c, ls='--', lw=1, alpha=0.7, label=f'+{thr}σ')
    ax.axhline(mu - thr*sigma, color=c, ls='--', lw=1, alpha=0.7, label=f'-{thr}σ')
    ax.set_title(f'threshold={thr}  →  전체 이상 포인트 {anom.sum()}개', fontsize=11)
    ax.legend(fontsize=8, loc='upper right')
    ax.set_ylabel('진폭')
axes[1].set_xlabel('시간 (s)')
plt.suptitle('과제 A: threshold 변경 비교 (처음 2,000 샘플 표시)', fontsize=13)
plt.tight_layout()
plt.show()

threshold=2: 이상 포인트 수 = 544
threshold=3: 이상 포인트 수 = 66
→ threshold를 낮추면 이상 포인트가 478개 더 감지됨

from scipy.fft import fft, fftfreq

fft_vals = fft(signal)
freqs = fftfreq(n, 1/sr)
pos = freqs > 0
f, mag = freqs[pos], np.abs(fft_vals[pos])

fig, axes = plt.subplots(1, 2, figsize=(14, 4))
axes[0].plot(f, mag, 'steelblue', lw=0.6)
axes[0].set_title('전체 FFT 스펙트럼 (실습)', fontsize=11)

axes[1].plot(f, mag, 'crimson', lw=0.8)
axes[1].set_xlim(0, 500)   # ← 과제 A: xlim 추가
axes[1].set_title('0~500Hz 확대 (과제 A: xlim 적용)', fontsize=11)
axes[1].set_xlabel('주파수 (Hz)')
# 피크 표시
peak_idx = np.argmax(mag)
axes[1].axvline(f[peak_idx], color='navy', ls='--', lw=1.2,
               label=f'최대: {f[peak_idx]:.1f}Hz')
axes[1].legend(fontsize=9)

plt.suptitle('과제 A: FFT xlim 적용 전후 비교', fontsize=13)
plt.tight_layout()
plt.show()

# 피처 생성
features = np.column_stack([
    signal,
    np.abs(signal),
    np.convolve(signal**2, np.ones(50)/50, mode='same')
])

results = {}
for cont in [0.05, 0.10]:
    iso = IsolationForest(contamination=cont, random_state=42, n_estimators=100)
    pred = iso.fit_predict(features)
    results[cont] = (pred == -1).sum()

print(f"contamination=0.05 → 이상 = {results[0.05]}개")
print(f"contamination=0.10 → 이상 = {results[0.10]}개")
print(f"→ 비율: {results[0.10]/results[0.05]:.1f}x 차이")

fig, ax = plt.subplots(figsize=(7, 4))
bars = ax.bar(['0.05 (엄격)', '0.10 (기본)'],
              [results[0.05], results[0.10]],
              color=['#4caf50','#f44336'], width=0.45)
ax.bar_label(bars, fontsize=12, padding=3)
ax.set_title('과제 A: contamination 변경 비교', fontsize=12)
ax.set_xlabel('contamination'); ax.set_ylabel('이상 샘플 수')
plt.tight_layout()
plt.show()

contamination=0.05 → 이상 = 500개
contamination=0.10 → 이상 = 1000개
→ 비율: 2.0x 차이

iso_main = IsolationForest(contamination=0.10, random_state=42, n_estimators=100)
scores = iso_main.fit(features).score_samples(features)
health_index = 100 * (1 - (scores - scores.min()) / (scores.max() - scores.min()))

below70 = np.where(health_index < 70)[0]
first_warn = below70[0] if len(below70) > 0 else None
print(f"경고 시작 인덱스: {first_warn}")
print(f"전체 샘플 중 경고 구간 비율: {len(below70)/len(health_index)*100:.1f}%")

fig, ax = plt.subplots(figsize=(12, 4))
ax.plot(health_index, 'steelblue', lw=0.8, label='Health Index')
ax.axhline(90, color='green',  ls='--', lw=1, label='정상 (≥90)')
